200 void Hist_Add(field_t *field)
201 {
202  int i;
203 
204  // Don't save blank lines in history.
205  if (!field->cursor)
206  {
207  return;
208  }
209 
210  assert(hist_count <= CON_HISTORY);
211  assert(hist_count >= 0);
212  assert(hist_current >= -1);
213  assert(hist_current <= hist_count);
214  // make some room
215  for (i = CON_HISTORY - 1; i > 0; i--)
216  {
217  ttyEditLines[i] = ttyEditLines[i - 1];
218  }
219  ttyEditLines[0] = *field;
220  if (hist_count < CON_HISTORY)
221  {
222  hist_count++;
223  }
224 
225  hist_current = -1; // re-init
226 }

288 ==================
289 CON_Init
290 
291 Initialize the console input (tty mode if possible)
292 ==================
293 */
294 void CON_Init(void)
295 {
296  struct termios tc;
297 
298  // If the process is backgrounded (running non interactively)
299  // then SIGTTIN or SIGTOU is emitted, if not caught, turns into a SIGSTP
300  signal(SIGTTIN, SIG_IGN);
301  signal(SIGTTOU, SIG_IGN);
302 
303  // If SIGCONT is received, reinitialize console
304  signal(SIGCONT, CON_SigCont);
305 
306  // Make stdin reads non-blocking
307  (void) fcntl(STDIN_FILENO, F_SETFL, fcntl(STDIN_FILENO, F_GETFL, 0) | O_NONBLOCK);
308 
309  if (!stdinIsATTY)
310  {
311  Com_Printf("tty console mode disabled\n");
312  ttycon_on = qfalse;
313  stdin_active = qtrue;
314  return;
315  }
316 
317  Field_Clear(&TTY_con);
318  tcgetattr(STDIN_FILENO, &TTY_tc);
319  TTY_erase = TTY_tc.c_cc[VERASE];
320  TTY_eof = TTY_tc.c_cc[VEOF];
321  tc = TTY_tc;
322 
323  /*
324  ECHO: don't echo input characters
325  ICANON: enable canonical mode. This enables the special
326  characters EOF, EOL, EOL2, ERASE, KILL, REPRINT,
327  STATUS, and WERASE, and buffers by lines.
328  ISIG: when any of the characters INTR, QUIT, SUSP, or
329  DSUSP are received, generate the corresponding signal
330  */
331  tc.c_lflag &= ~(ECHO | ICANON);
332 
333  /*
334  ISTRIP strip off bit 8
335  INPCK enable input parity checking
336  */
337  tc.c_iflag &= ~(ISTRIP | INPCK);
338  tc.c_cc[VMIN] = 1;
339  tc.c_cc[VTIME] = 0;
340  tcsetattr(STDIN_FILENO, TCSADRAIN, &tc);
341  ttycon_on = qtrue;
342  ttycon_hide = 1; // Mark as hidden, so prompt is shown in CON_Show
343  CON_Show();
344 }

351 char *CON_Input(void)
352 {
353  // we use this when sending back commands
354  static char text[MAX_EDIT_LINE];
355  char key;
356  field_t *history;
357 
358  if (ttycon_on)
359  {
360  size_t UNUSED_VAR size;
361  int avail = read(STDIN_FILENO, &key, 1);
362 
363  if (avail != -1)
364  {
365  // we have something
366  // backspace?
367  // NOTE TTimo testing a lot of values .. seems it's the only way to get it to work everywhere
368  if ((key == TTY_erase) || (key == 127) || (key == 8))
369  {
370  if (TTY_con.cursor > 0)
371  {
372  TTY_con.cursor--;
373  TTY_con.buffer[TTY_con.cursor] = '\0';
374  CON_Back();
375  }
376 
377  return NULL;
378  }
379  // check if this is a control char
380  if ((key) && (key) < ' ')
381  {
382  if (key == '\n')
383  {
384 #ifndef DEDICATED
385  // if not in the game explicitly prepend a slash if needed
386  if (clc.state != CA_ACTIVE && TTY_con.cursor &&
387  TTY_con.buffer[0] != '/' && TTY_con.buffer[0] != '\\')
388  {
389  memmove(TTY_con.buffer + 1, TTY_con.buffer, sizeof(TTY_con.buffer) - 1);
390  TTY_con.buffer[0] = '\\';
391  TTY_con.cursor++;
392  }
393 
394  if (TTY_con.buffer[0] == '/' || TTY_con.buffer[0] == '\\')
395  {
396  Q_strncpyz(text, TTY_con.buffer + 1, sizeof(text));
397  }
398  else if (TTY_con.cursor)
399  {
400  Com_sprintf(text, sizeof(text), "cmd say %s", TTY_con.buffer);
401  }
402  else
403  {
404  text[0] = '\0';
405  }
406 
407  // push it in history
408  Hist_Add(&TTY_con);
409  CON_Hide();
410  Com_Printf("%s%s\n", TTY_CONSOLE_PROMPT, TTY_con.buffer);
411  Field_Clear(&TTY_con);
412  CON_Show();
413 #else
414  // push it in history
415  Hist_Add(&TTY_con);
416  Q_strncpyz(text, TTY_con.buffer, sizeof(text));
417  Field_Clear(&TTY_con);
418  key = '\n';
419  size = write(STDOUT_FILENO, &key, 1);
420  size = write(STDOUT_FILENO, TTY_CONSOLE_PROMPT, strlen(TTY_CONSOLE_PROMPT));
421 #endif
422 
423  return text;
424  }
425  if (key == '\t')
426  {
427  CON_Hide();
428  Field_AutoComplete(&TTY_con);
429  CON_Show();
430  return NULL;
431  }
432  avail = read(STDIN_FILENO, &key, 1);
433  if (avail != -1)
434  {
435  // VT 100 keys
436  if (key == '[' || key == 'O')
437  {
438  avail = read(STDIN_FILENO, &key, 1);
439  if (avail != -1)
440  {
441  switch (key)
442  {
443  case 'A':
444  history = Hist_Prev();
445  if (history)
446  {
447  CON_Hide();
448  TTY_con = *history;
449  CON_Show();
450  }
451  tcflush(STDIN_FILENO, TCIFLUSH);
452  return NULL;
453  break;
454  case 'B':
455  history = Hist_Next();
456  CON_Hide();
457  if (history)
458  {
459  TTY_con = *history;
460  }
461  else
462  {
463  Field_Clear(&TTY_con);
464  }
465  CON_Show();
466  tcflush(STDIN_FILENO, TCIFLUSH);
467  return NULL;
468  break;
469  case 'C':
470  return NULL;
471  case 'D':
472  return NULL;
473  }
474  }
475  }
476  }
477 
478  Com_DPrintf("droping ISCTL sequence: %d, TTY_erase: %d\n", key, TTY_erase);
479  tcflush(STDIN_FILENO, TCIFLUSH);
480  return NULL;
481  }
482  if (TTY_con.cursor >= sizeof(text) - 1)
483  {
484  return NULL;
485  }
486 
487  // push regular character
488  TTY_con.buffer[TTY_con.cursor] = key;
489  TTY_con.cursor++; // next char will always be '\0'
490  // print the current line (this is differential)
491  size = write(STDOUT_FILENO, &key, 1);
492  }
493 
494  return NULL;
495  }
496  else if (stdin_active)
497  {
498  int len;
499  fd_set fdset;
500  struct timeval timeout;
501 
502  FD_ZERO(&fdset);
503  FD_SET(STDIN_FILENO, &fdset); // stdin
504  timeout.tv_sec = 0;
505  timeout.tv_usec = 0;
506  if (select(STDIN_FILENO + 1, &fdset, NULL, NULL, &timeout) == -1 || !FD_ISSET(STDIN_FILENO, &fdset))
507  {
508  return NULL;
509  }
510 
511  len = read(STDIN_FILENO, text, sizeof(text));
512  if (len == 0) // eof!
513  {
514  stdin_active = qfalse;
515  return NULL;
516  }
517 
518  if (len < 1)
519  {
520  return NULL;
521  }
522 
523  text[len - 1] = 0; // rip off the /n and terminate
524 
525  return text;
526  }
527 
528  return NULL;
529 }
