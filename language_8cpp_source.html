380 Language
381 Language::from_spec(const std::string& language, const std::string& country, const std::string& modifier)
382 {
383 #if __cplusplus >= 201103L // C++11
384  typedef std::unordered_map<std::string, std::vector<const LanguageSpec *> > LanguageSpecMap;
385 #else
386  typedef std::map<std::string, std::vector<const LanguageSpec *> > LanguageSpecMap;
387 #endif
388  static LanguageSpecMap language_map;
389 
390  if (language_map.empty())
391  { // Init language_map
392  for (int i = 0; languages[i].language != NULL; ++i)
393  language_map[languages[i].language].push_back(&languages[i]);
394  }
395 
396  LanguageSpecMap::iterator i = language_map.find(language);
397  if (i != language_map.end())
398  {
399  std::vector<const LanguageSpec *>& lst = i->second;
400 
401  LanguageSpec tmpspec;
402  tmpspec.language = language.c_str();
403  tmpspec.country = country.c_str();
404  tmpspec.modifier = modifier.c_str();
405  Language tmplang(&tmpspec);
406 
407  const LanguageSpec *best_match = 0;
408  int best_match_score = 0;
409  for (std::vector<const LanguageSpec *>::iterator j = lst.begin(); j != lst.end(); ++j)
410  { // Search for the language that best matches the given spec, value country more then modifier
411  int score = Language::match(Language(*j), tmplang);
412 
413  if (score > best_match_score)
414  {
415  best_match = *j;
416  best_match_score = score;
417  }
418  }
419  assert(best_match);
420  return Language(best_match);
421  }
422  else
423  {
424  return Language();
425  }
426 }

434 Language
435 Language::from_env(const std::string& env)
436 {
437  // Split LANGUAGE_COUNTRY.CODESET@MODIFIER into parts
438  std::string::size_type ln = env.find('_');
439  std::string::size_type dt = env.find('.');
440  std::string::size_type at = env.find('@');
441 
442  std::string language;
443  std::string country;
444  std::string codeset;
445  std::string modifier;
446 
447  //std::cout << ln << " " << dt << " " << at << std::endl;
448 
449  language = env.substr(0, std::min(std::min(ln, dt), at));
450 
451  if (ln != std::string::npos && ln + 1 < env.size()) // _
452  {
453  country = env.substr(ln + 1, (std::min(dt, at) == std::string::npos) ? std::string::npos : std::min(dt, at) - (ln + 1));
454  }
455 
456  if (dt != std::string::npos && dt + 1 < env.size()) // .
457  {
458  codeset = env.substr(dt + 1, (at == std::string::npos) ? std::string::npos : (at - (dt + 1)));
459  }
460 
461  if (at != std::string::npos && at + 1 < env.size()) // @
462  {
463  modifier = env.substr(at + 1);
464  }
465 
466  return from_spec(language, country, modifier);
467 }

479 int
480 Language::match(const Language& lhs, const Language& rhs)
481 {
482  if (lhs.get_language() != rhs.get_language())
483  {
484  return 0;
485  }
486  else
487  {
488  static int match_tbl[3][3] = {
489  // modifier match, wildchard, miss
490  { 9, 8, 5 }, // country match
491  { 7, 6, 3 }, // country wildcard
492  { 4, 2, 1 }, // country miss
493  };
494 
495  int c;
496  if (lhs.get_country() == rhs.get_country())
497  {
498  c = 0;
499  }
500  else if (lhs.get_country().empty() || rhs.get_country().empty())
