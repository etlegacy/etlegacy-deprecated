49 static qboolean ParseVector(char **text, int count, float *v)
50 {
51  char *token;
52  int i;
53 
54  token = COM_ParseExt(text, qfalse);
55  if (strcmp(token, "("))
56  {
57  Ren_Warning("WARNING: missing parenthesis '(' in shader '%s' of token '%s'\n", shader.name, token);
58  return qfalse;
59  }
60 
61  for (i = 0; i < count; i++)
62  {
63  token = COM_ParseExt(text, qfalse);
64  if (!token[0])
65  {
66  Ren_Warning("WARNING: missing vector element in shader '%s' - no token\n", shader.name);
67  return qfalse;
68  }
69  v[i] = atof(token);
70  }
71 
72  token = COM_ParseExt(text, qfalse);
73  if (strcmp(token, ")"))
74  {
75  Ren_Warning("WARNING: missing parenthesis ')' in shader '%s' of token '%s'\n", shader.name, token);
76  return qfalse;
77  }
78 
79  return qtrue;
80 }
