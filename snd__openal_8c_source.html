439 static void S_AL_BufferLoad(sfxHandle_t sfx, qboolean cache)
440 {
441  ALenum error;
442  ALuint format;
443 
444  void *data;
445  snd_info_t info;
446  alSfx_t *curSfx = &knownSfx[sfx];
447 
448  // Nothing?
449  if (curSfx->filename[0] == '\0')
450  {
451  return;
452  }
453 
454  // Player SFX
455  if (curSfx->filename[0] == '*')
456  {
457  return;
458  }
459 
460  // Already done?
461  if ((curSfx->inMemory) || (curSfx->isDefault) || (!cache && curSfx->isDefaultChecked))
462  {
463  return;
464  }
465 
466  // Try to load
467  data = S_CodecLoad(curSfx->filename, &info);
468  if (!data)
469  {
470  S_AL_BufferUseDefault(sfx);
471  Com_Printf(S_COLOR_RED "ERROR S_AL_BufferLoad: S_CodecLoad failed for \"%s\"\n", curSfx->filename);
472  return;
473  }
474 
475  curSfx->isDefaultChecked = qtrue;
476 
477  if (!cache)
478  {
479  // Don't create AL cache
480  Hunk_FreeTempMemory(data);
481  return;
482  }
483 
484  format = S_AL_Format(info.width, info.channels);
485 
486  // Create a buffer
487  S_AL_ClearError(qfalse);
488  qalGenBuffers(1, &curSfx->buffer);
489  if ((error = qalGetError()) != AL_NO_ERROR)
490  {
491  S_AL_BufferUseDefault(sfx);
492  Hunk_FreeTempMemory(data);
493  Com_Printf(S_COLOR_RED "ERROR S_AL_BufferLoad: Can't create a sound buffer for %s - %s\n",
494  curSfx->filename, S_AL_ErrorMsg(error));
495  return;
496  }
497 
498  // Fill the buffer
499  if (info.size == 0)
500  {
501  // We have no data to buffer, so buffer silence
502  byte dummyData[2] = { 0 };
503 
504  qalBufferData(curSfx->buffer, AL_FORMAT_MONO16, (void *)dummyData, 2, 22050);
505  }
506  else
507  {
508  qalBufferData(curSfx->buffer, format, data, info.size, info.rate);
509  }
510 
511  error = qalGetError();
512 
513  // If we ran out of memory, start evicting the least recently used sounds
514  while (error == AL_OUT_OF_MEMORY)
515  {
516  if (!S_AL_BufferEvict())
517  {
518  qalDeleteBuffers(1, &curSfx->buffer);
519  S_AL_BufferUseDefault(sfx);
520  Hunk_FreeTempMemory(data);
521  Com_Printf(S_COLOR_RED "ERROR S_AL_BufferLoad: Out of memory loading %s\n", curSfx->filename);
522  return;
523  }
524 
525  // Try load it again
526  qalBufferData(curSfx->buffer, format, data, info.size, info.rate);
527  error = qalGetError();
528  }
529 
530  // Some other error condition
531  if (error != AL_NO_ERROR)
532  {
533  qalDeleteBuffers(1, &curSfx->buffer);
534  S_AL_BufferUseDefault(sfx);
535  Hunk_FreeTempMemory(data);
536  Com_Printf(S_COLOR_RED "ERROR S_AL_BufferLoad: Can't fill sound buffer for %s - %s\n",
537  curSfx->filename, S_AL_ErrorMsg(error));
538  return;
539  }
540 
541  curSfx->info = info;
542 
543  // Free the memory
544  Hunk_FreeTempMemory(data);
545 
546  // Woo!
547  curSfx->inMemory = qtrue;
548 }

774 static void S_AL_ScaleGain(src_t *chksrc, vec3_t origin)
775 {
776  float distance = 0.0f;
777 
778  if (!chksrc->local)
779  {
780  distance = vec3_distance(origin, lastListenerOrigin);
781  }
782 
783  // If we exceed a certain distance, scale the gain linearly until the sound
784  // vanishes into nothingness.
785  if (!chksrc->local && (distance -= s_alMaxDistance->value) > 0)
786  {
787  float scaleFactor;
788 
789  if (distance >= s_alGraceDistance->value)
790  {
791  scaleFactor = 0.0f;
792  }
793  else
794  {
795  scaleFactor = 1.0f - distance / s_alGraceDistance->value;
796  }
797

799 
800  if (chksrc->scaleGain != scaleFactor)
801  {
802  chksrc->scaleGain = scaleFactor;
803  S_AL_Gain(chksrc->alSource, chksrc->scaleGain);
804  }
805  }
