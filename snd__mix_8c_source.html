145 void S_TransferStereo16(unsigned long *pbuf, int endtime)
146 {
147  int lpos;
148  int ls_paintedtime = s_paintedtime;
149 
150  snd_p = (int *) paintbuffer;
151 
152  while (ls_paintedtime < endtime)
153  {
154  // handle recirculating buffer issues
155  lpos = ls_paintedtime & ((dma.samples >> 1) - 1);
156 
157  snd_out = (short *) pbuf + (lpos << 1);
158 
159  snd_linear_count = (dma.samples >> 1) - lpos;
160  if (ls_paintedtime + snd_linear_count > endtime)
161  {
162  snd_linear_count = endtime - ls_paintedtime;
163  }
164 
165  snd_linear_count <<= 1;
166 
167  // write a linear blast of samples
168  S_WriteLinearBlastStereo16();
169 
170  snd_p += snd_linear_count;
171  ls_paintedtime += (snd_linear_count >> 1);
172 
173  if (CL_VideoRecording())
174  {
175  CL_WriteAVIAudioFrame((byte *)snd_out, snd_linear_count << 1);
176  }
177  }

184 void S_TransferPaintBuffer(int endtime)
185 {
186  unsigned long *pbuf = (unsigned long *)dma.buffer;
187 
188  if (s_testsound->integer)
189  {
190  int i;
191 
192  // write a fixed sine wave
193  for (i = 0 ; i < (endtime - s_paintedtime) ; i++)
194  {
195  paintbuffer[i].left = paintbuffer[i].right = sin((s_paintedtime + i) * 0.1) * 20000 * 256;
196  }
197  }
198 
199  if (dma.samplebits == 16 && dma.channels == 2) // optimized case
200  {
201  S_TransferStereo16(pbuf, endtime);
202  }
203  else // general case
204  {
205  int out_mask = dma.samples - 1;
206  int count = (endtime - s_paintedtime) * dma.channels;
207  int out_idx = s_paintedtime * dma.channels & out_mask;
208  int step = 3 - dma.channels;
209  int val;
210  int *p = (int *) paintbuffer;
211 
212  if (dma.samplebits == 16)
213  {
214  short *out = (short *) pbuf;
215 
216  while (count--)
217  {
218  val = *p >> 8;
219  p += step;
220  if (val > 0x7fff)
221  {
222  val = 0x7fff;
223  }
224  else if (val < -32768)
225  {
226  val = -32768;
227  }
228  out[out_idx] = val;
229  out_idx = (out_idx + 1) & out_mask;
230  }
231  }
232  else if (dma.samplebits == 8)
233  {
234  unsigned char *out = (unsigned char *) pbuf;
235 
236  while (count--)
237  {
238  val = *p >> 8;
239  p += step;
240  if (val > 0x7fff)
241  {
242  val = 0x7fff;
243  }
244  else if (val < -32768)
245  {
246  val = -32768;
247  }
248  out[out_idx] = (val >> 8) + 128;
249  out_idx = (out_idx + 1) & out_mask;
250  }
251  }
252  }

270 static void S_PaintChannelFrom16_altivec(channel_t *ch, const sfx_t *sc, int count, int sampleOffset, int bufferOffset)
271 {
272  int data, aoff, boff;
273  int leftvol, rightvol;
274  int i, j;
275  portable_samplepair_t *samp = &paintbuffer[bufferOffset];
276  sndBuffer *chunk;
277  short *samples;
278  float ooff, fdata[2], fdiv, fleftvol, frightvol;
279 
280  if (sc->soundChannels <= 0)
281  {
282  return;
283  }
284 
285  if (ch->doppler)
286  {
287  sampleOffset = sampleOffset * ch->oldDopplerScale;
288  }
289 
290  if (sc->soundChannels == 2)
291  {
292  sampleOffset *= sc->soundChannels;
293 
294  if (sampleOffset & 1)
295  {
296  sampleOffset &= ~1;
297  }
298  }
299 
300  chunk = sc->soundData;
301  while (sampleOffset >= SND_CHUNK_SIZE)
302  {
303  chunk = chunk->next;
304  sampleOffset -= SND_CHUNK_SIZE;
305  if (!chunk)
306  {
307  chunk = sc->soundData;
308  }
309  }
310 
311  if (!ch->doppler || ch->dopplerScale == 1.0f)
312  {
313  vector signed short volume_vec;
314  vector unsigned int volume_shift;
315  int vectorCount, samplesLeft, chunkSamplesLeft;
316  leftvol = ch->leftvol * snd_vol;
317  rightvol = ch->rightvol * snd_vol;
318  samples = chunk->sndChunk;
319  ((short *)&volume_vec)[0] = leftvol;
320  ((short *)&volume_vec)[1] = leftvol;
321  ((short *)&volume_vec)[4] = leftvol;
322  ((short *)&volume_vec)[5] = leftvol;
323  ((short *)&volume_vec)[2] = rightvol;
324  ((short *)&volume_vec)[3] = rightvol;
325  ((short *)&volume_vec)[6] = rightvol;
326  ((short *)&volume_vec)[7] = rightvol;
327  volume_shift = vec_splat_u32(8);
328  i = 0;
329 
330  while (i < count)
331  {
332  /* Try to align destination to 16-byte boundary */
333  while (i < count && (((unsigned long)&samp[i] & 0x1f) || ((count - i) < 8) || ((SND_CHUNK_SIZE - sampleOffset) < 8)))
334  {
335  data = samples[sampleOffset++];
336  samp[i].left += (data * leftvol) >> 8;
337  if (sc->soundChannels == 2)
338  {
339  data = samples[sampleOffset++];
340  }
341  samp[i].right += (data * rightvol) >> 8;
342 
343  if (sampleOffset == SND_CHUNK_SIZE)
344  {
345  chunk = chunk->next;
346  samples = chunk->sndChunk;
347  sampleOffset = 0;
348  }
349  i++;
350  }
351  /* Destination is now aligned. Process as many 8-sample
352  chunks as we can before we run out of room from the current
353  sound chunk. We do 8 per loop to avoid extra source data reads. */
354  samplesLeft = count - i;
355  chunkSamplesLeft = SND_CHUNK_SIZE - sampleOffset;
356  if (samplesLeft > chunkSamplesLeft)
357  {
358  samplesLeft = chunkSamplesLeft;
359  }
360 
361  vectorCount = samplesLeft / 8;
362 
363  if (vectorCount)
364  {
365  vector unsigned char tmp;
366  vector short s0, s1, sampleData0, sampleData1;
367  vector signed int merge0, merge1;
368  vector signed int d0, d1, d2, d3;
369  vector unsigned char samplePermute0 =
370  VECCONST_UINT8(0, 1, 4, 5, 0, 1, 4, 5, 2, 3, 6, 7, 2, 3, 6, 7);
371  vector unsigned char samplePermute1 =
372  VECCONST_UINT8(8, 9, 12, 13, 8, 9, 12, 13, 10, 11, 14, 15, 10, 11, 14, 15);
373  vector unsigned char loadPermute0, loadPermute1;
374 
375  // Rather than permute the vectors after we load them to do the sample
376  // replication and rearrangement, we permute the alignment vector so
377  // we do everything in one step below and avoid data shuffling.
378  tmp = vec_lvsl(0, &samples[sampleOffset]);
379  loadPermute0 = vec_perm(tmp, tmp, samplePermute0);
380  loadPermute1 = vec_perm(tmp, tmp, samplePermute1);
381 
382  s0 = *(vector short *)&samples[sampleOffset];
383  while (vectorCount)
384  {
385  /* Load up source (16-bit) sample data */
386  s1 = *(vector short *)&samples[sampleOffset + 7];
387 
388  /* Load up destination sample data */
389  d0 = *(vector signed int *)&samp[i];
390  d1 = *(vector signed int *)&samp[i + 2];
391  d2 = *(vector signed int *)&samp[i + 4];
392  d3 = *(vector signed int *)&samp[i + 6];
393 
394  sampleData0 = vec_perm(s0, s1, loadPermute0);
395  sampleData1 = vec_perm(s0, s1, loadPermute1);
396 
397  merge0 = vec_mule(sampleData0, volume_vec);
398  merge0 = vec_sra(merge0, volume_shift); /* Shift down to proper range */
399 
400  merge1 = vec_mulo(sampleData0, volume_vec);
401  merge1 = vec_sra(merge1, volume_shift);
402 
403  d0 = vec_add(merge0, d0);
404  d1 = vec_add(merge1, d1);
405 
406  merge0 = vec_mule(sampleData1, volume_vec);
407  merge0 = vec_sra(merge0, volume_shift); /* Shift down to proper range */
408 
409  merge1 = vec_mulo(sampleData1, volume_vec);
410  merge1 = vec_sra(merge1, volume_shift);
411 
412  d2 = vec_add(merge0, d2);
413  d3 = vec_add(merge1, d3);
414 
415  /* Store destination sample data */
416  *(vector signed int *)&samp[i] = d0;
417  *(vector signed int *)&samp[i + 2] = d1;
418  *(vector signed int *)&samp[i + 4] = d2;
419  *(vector signed int *)&samp[i + 6] = d3;
420 
421  i += 8;
422  vectorCount--;
423  s0 = s1;
424  sampleOffset += 8;
425  }
426  if (sampleOffset == SND_CHUNK_SIZE)
427  {
428  chunk = chunk->next;
429  samples = chunk->sndChunk;
430  sampleOffset = 0;
431  }
432  }
433  }
434  }
435  else
436  {
437  fleftvol = ch->leftvol * snd_vol;
438  frightvol = ch->rightvol * snd_vol;
439 
440  ooff = sampleOffset;
441  samples = chunk->sndChunk;
442 
443  for (i = 0 ; i < count ; i++)
444  {
445 
446  aoff = ooff;
447  ooff = ooff + ch->dopplerScale * sc->soundChannels;
448  boff = ooff;
449  fdata[0] = fdata[1] = 0;
450  for (j = aoff; j < boff; j += sc->soundChannels)
451  {
452  if (j == SND_CHUNK_SIZE)
453  {
454  chunk = chunk->next;
455  if (!chunk)
456  {
457  chunk = sc->soundData;
458  }
459  samples = chunk->sndChunk;
460  ooff -= SND_CHUNK_SIZE;
461  }
462  if (sc->soundChannels == 2)
463  {
464  fdata[0] += samples[j & (SND_CHUNK_SIZE - 1)];
465  fdata[1] += samples[(j + 1) & (SND_CHUNK_SIZE - 1)];
466  }
467  else
468  {
469  fdata[0] += samples[j & (SND_CHUNK_SIZE - 1)];
470  fdata[1] += samples[j & (SND_CHUNK_SIZE - 1)];
471  }
472  }
473  fdiv = 256 * (boff - aoff) / sc->soundChannels;
474  samp[i].left += (fdata[0] * fleftvol) / fdiv;
475  samp[i].right += (fdata[1] * frightvol) / fdiv;
476  }
477  }
478 }
