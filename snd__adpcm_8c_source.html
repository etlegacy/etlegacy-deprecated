68 void S_AdpcmEncode(short indata[], char outdata[], int len, struct adpcm_state *state)
69 {
70  short *inp = indata; // Input buffer pointer
71  signed char *outp = (signed char *)outdata; // output buffer pointer
72  int val; // Current input sample value
73  int sign; // Current adpcm sign bit
74  int delta; // Current adpcm output value
75  int diff; // Difference between val and sample
76  int index = state->index; // Current step change index
77  int step = stepsizeTable[index]; // Stepsize
78  int valpred = state->sample; // Predicted output value
79  int vpdiff; // Current change to valpred
80  int outputbuffer = 0; // place to keep previous 4-bit value
81  int bufferstep = 1; // toggle between outputbuffer/output
82 
83  for ( ; len > 0 ; len--)
84  {
85  val = *inp++;
86 
87  /* Step 1 - compute difference with previous value */
88  diff = val - valpred;
89  sign = (diff < 0) ? 8 : 0;
90  if (sign)
91  {
92  diff = (-diff);
93  }
94 
95  /* Step 2 - Divide and clamp */
96  /* Note:
97  ** This code *approximately* computes:
98  ** delta = diff*4/step;
99  ** vpdiff = (delta+0.5)*step/4;
100  ** but in shift step bits are dropped. The net result of this is
101  ** that even if you have fast mul/div hardware you cannot put it to
102  ** good use since the fixup would be too expensive.
103  */
104  delta = 0;
105  vpdiff = (step >> 3);
106 
107  if (diff >= step)
108  {
109  delta = 4;
110  diff -= step;
111  vpdiff += step;
112  }
113  step >>= 1;
114  if (diff >= step)
115  {
116  delta |= 2;
117  diff -= step;
118  vpdiff += step;
119  }
120  step >>= 1;
121  if (diff >= step)
122  {
123  delta |= 1;
124  vpdiff += step;
125  }
126 
127  /* Step 3 - Update previous value */
128  if (sign)
129  {
130  valpred -= vpdiff;
131  }
132  else
133  {
134  valpred += vpdiff;
135  }
136 
137  /* Step 4 - Clamp previous value to 16 bits */
138  if (valpred > 32767)
139  {
140  valpred = 32767;
141  }
142  else if (valpred < -32768)
143  {
144  valpred = -32768;
145  }
146 
147  /* Step 5 - Assemble value, update index and step values */
148  delta |= sign;
149 
150  index += indexTable[delta];
151  if (index < 0)
152  {
153  index = 0;
154  }
155  if (index > 88)
156  {
157  index = 88;
158  }
159  step = stepsizeTable[index];
160 
161  /* Step 6 - Output value */
162  if (bufferstep)
163  {
164  outputbuffer = (delta << 4) & 0xf0;
165  }
166  else
167  {
168  *outp++ = (delta & 0x0f) | outputbuffer;
169  }
170  bufferstep = !bufferstep;
171  }
172 
173  /* Output last step, if needed */
174  if (!bufferstep)
175  {
176  *outp++ = outputbuffer;
177  }
178 
179  state->sample = valpred;
180  state->index = index;
181 }

190 static void S_AdpcmDecode(const char indata[], short *outdata, int len, struct adpcm_state *state)
191 {
192  signed char *inp = (signed char *)indata; // Input buffer pointer
193  int outp = 0; // output buffer pointer
194  int sign; // Current adpcm sign bit
195  int delta; // Current adpcm output value
196  int index = state->index; // Current step change index
197  int step = stepsizeTable[index]; // Stepsize
198  int valpred = state->sample; // Predicted value
199  int vpdiff; // Current change to valpred
200  int inputbuffer = 0; // place to keep next 4-bit value
201  int bufferstep = 0; // toggle between inputbuffer/input
202 
203  for ( ; len > 0 ; len--)
204  {
205  /* Step 1 - get the delta value */
206  if (bufferstep)
207  {
208  delta = inputbuffer & 0xf;
209  }
210  else
211  {
212  inputbuffer = *inp++;
213  delta = (inputbuffer >> 4) & 0xf;
214  }
215  bufferstep = !bufferstep;
216 
217  /* Step 2 - Find new index value (for later) */
218  index += indexTable[delta];
219  if (index < 0)
220  {
221  index = 0;
222  }
223  if (index > 88)
224  {
225  index = 88;
226  }
227 
228  /* Step 3 - Separate sign and magnitude */
229  sign = delta & 8;
230  delta = delta & 7;
231 
232  /* Step 4 - Compute difference and new predicted value */
233  /*
234  ** Computes 'vpdiff = (delta+0.5)*step/4', but see comment
235  ** in adpcm_coder.
236  */
237  vpdiff = step >> 3;
238  if (delta & 4)
239  {
240  vpdiff += step;
241  }
242  if (delta & 2)
243  {
244  vpdiff += step >> 1;
245  }
246  if (delta & 1)
247  {
248  vpdiff += step >> 2;
249  }
250 
251  if (sign)
252  {
253  valpred -= vpdiff;
254  }
255  else
256  {
257  valpred += vpdiff;
258  }
259 
260  /* Step 5 - clamp output value */
261  if (valpred > 32767)
262  {
263  valpred = 32767;
264  }
265  else if (valpred < -32768)
266  {
267  valpred = -32768;
268  }
269 
270  /* Step 6 - Update step value */
271  step = stepsizeTable[index];
272 
273  /* Step 7 - Output value */
274  outdata[outp] = valpred;
275  outp++;
276  }
277 
278  state->sample = valpred;
279  state->index = index;
280 }
