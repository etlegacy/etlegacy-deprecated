190 static ID_INLINE void END_CHUNK(void)
191 {

502 void CL_WriteAVIVideoFrame(const byte *imageBuffer, int size)
503 {
504  int chunkOffset = afd.fileSize - afd.moviOffset - 8;
505  int chunkSize = 8 + size;
506  int paddingSize = PAD(size, 2) - size;
507  byte padding[4] = { 0 };
508 
509  if (!afd.fileOpen)
510  {
511  return;
512  }
513 
514  // Chunk header + contents + padding
515  if (CL_CheckFileSize(8 + size + 2))
516  {
517  return;
518  }
519 
520  bufIndex = 0;
521  WRITE_STRING("00dc");
522  WRITE_4BYTES(size);
523 
524  SafeFS_Write(buffer, 8, afd.f);
525  SafeFS_Write(imageBuffer, size, afd.f);
526  SafeFS_Write(padding, paddingSize, afd.f);
527  afd.fileSize += (chunkSize + paddingSize);
528 
529  afd.numVideoFrames++;
530  afd.moviSize += (chunkSize + paddingSize);
531 
532  if (size > afd.maxRecordSize)
533  {
534  afd.maxRecordSize = size;
535  }
536 
537  // Index
538  bufIndex = 0;
539  WRITE_STRING("00dc"); //dwIdentifier
540  WRITE_4BYTES(0x00000010); //dwFlags (all frames are KeyFrames)
541  WRITE_4BYTES(chunkOffset); //dwOffset
542  WRITE_4BYTES(size); //dwLength
543  SafeFS_Write(buffer, 16, afd.idxF);
544 
545  afd.numIndices++;
546 }

555 void CL_WriteAVIAudioFrame(const byte *pcmBuffer, int size)

563  }

566  {
567  return;
568  }
569

587  {

589  int chunkSize = 8 + bytesInBuffer;

591  byte padding[4] = { 0 };

596

638 qboolean CL_CloseAVI(void)
639 {
640  int indexRemainder;
641  int indexSize = afd.numIndices * 16;
642  const char *idxFileName = va("%s" INDEX_FILE_EXTENSION, afd.fileName);
643 
644  // AVI file isn't open

648  }

651

653  bufIndex = 0;

655  SafeFS_Write(buffer, bufIndex, afd.idxF);
656  FS_FCloseFile(afd.idxF);
657 
658  Z_Free(afd.cBuffer);
659  Z_Free(afd.eBuffer);
660 
661  // Write index
662 
663  // Open the temp index file
664  if ((indexSize = FS_FOpenFileRead(idxFileName, &afd.idxF, qtrue)) <= 0)
665  {
666  FS_FCloseFile(afd.f);
667  return qfalse;
668  }
669 
670  indexRemainder = indexSize;
671 
672  // Append index to end of avi file
673  while (indexRemainder > MAX_AVI_BUFFER)
674  {
675  FS_Read(buffer, MAX_AVI_BUFFER, afd.idxF);
676  SafeFS_Write(buffer, MAX_AVI_BUFFER, afd.f);
677  afd.fileSize += MAX_AVI_BUFFER;
678  indexRemainder -= MAX_AVI_BUFFER;
679  }
680  FS_Read(buffer, indexRemainder, afd.idxF);
681  SafeFS_Write(buffer, indexRemainder, afd.f);
682  afd.fileSize += indexRemainder;
683  FS_FCloseFile(afd.idxF);
684 
685  // Remove temp index file
686  FS_HomeRemove(idxFileName);
687 
688  // Write the real header
689  (void) FS_Seek(afd.f, 0, FS_SEEK_SET);
690  CL_WriteAVIHeader();
691 
692  bufIndex = 4;
693  WRITE_4BYTES(afd.fileSize - 8); // "RIFF" size
694 
695  bufIndex = afd.moviOffset + 4; // Skip "LIST"
696  WRITE_4BYTES(afd.moviSize);
697 
698  SafeFS_Write(buffer, bufIndex, afd.f);
699 
700  FS_FCloseFile(afd.f);
701 
702  Com_Printf("Wrote %d:%d frames to %s\n", afd.numVideoFrames, afd.numAudioFrames, afd.fileName);
703

705 }
