41  MODEL TESTING
42 
43 The viewthing and gun positioning tools from Q2 have been integrated and
44 enhanced into a single model testing facility.
45 
46 Model viewing can begin with either "testmodel <modelname>" or "testgun <modelname>".
47 
48 The names must be the full pathname after the basedir, like
49 "models/weapons/v_launch/tris.md3" or "players/male/tris.md3"
50 
51 Testmodel will create a fake entity 100 units in front of the current view
52 position, directly facing the viewer. It will remain immobile, so you can
53 move around it to view it from different angles.
54 
55 Testgun will cause the model to follow the player around and supress the real
56 view weapon model. The default frame 0 of most guns is completely off screen,
57 so you will probably have to cycle a couple frames to see it.
58 
59 "nextframe", "prevframe", "nextskin", and "prevskin" commands will change the
60 frame or skin of the testmodel. These are bound to F5, F6, F7, and F8 in
61 default.cfg see CONFIG_NAME_DEFAULT
62 
63 If a gun is being tested, the "gun_x", "gun_y", and "gun_z" variables will let
64 you adjust the positioning.
65 
66 Note that none of the model testing features update while the game is paused, so
67 it may be convenient to test with deathmatch set to 1 so that bringing down the
68 console doesn't pause the game.

329 
330  VectorCopy(cg.refdefViewAngles, focusAngles);
331 
332  // If dead, look at medic or allow freelook if none in range
333  if (cg.predictedPlayerState.stats[STAT_HEALTH] <= 0
334  && !(cg.predictedPlayerState.pm_flags & PMF_LIMBO))
335  {
336  // Force yaw to 0 if we're tracking a medic
337  if (cg.snap->ps.viewlocked != VIEWLOCK_MEDIC)
338  {
339  // Do short2angle AFTER the network part
340  focusAngles[YAW] = SHORT2ANGLE(cg.predictedPlayerState.stats[STAT_DEAD_YAW]);
341  cg.refdefViewAngles[YAW] = SHORT2ANGLE(cg.predictedPlayerState.stats[STAT_DEAD_YAW]);
342  }
343  }
344 
345  if (focusAngles[PITCH] > 45)
346  {
347  focusAngles[PITCH] = 45; // don't go too far overhead
348  }
349  AngleVectors(focusAngles, forward, NULL, NULL);
350 
351  if (cg_thirdPerson.integer == 2)
352  {
353  VectorCopy(cg.predictedPlayerState.origin, focusPoint);
354  }
355  else
356  {
357  VectorMA(cg.refdef_current->vieworg, FOCUS_DISTANCE, forward, focusPoint);
358  }
359 
360  VectorCopy(cg.refdef_current->vieworg, view);
361 
362  view[2] += 8;

364 
365  AngleVectors(cg.refdefViewAngles, forward, right, up);

368  sideScale = sin(cg_thirdPersonAngle.value / 180 * M_PI);
369  VectorMA(view, -cg_thirdPersonRange.value * forwardScale, forward, view);
370  VectorMA(view, -cg_thirdPersonRange.value * sideScale, right, view);
371 
372  // trace a ray from the origin to the viewpoint to make sure the view isn't
373  // in a solid block. Use an 8 by 8 block to prevent the view from near clipping anything
374 
375  CG_Trace(&trace, cg.refdef_current->vieworg, mins, maxs, view, cg.predictedPlayerState.clientNum, MASK_SOLID);
376 
377  if (trace.fraction != 1.0f)
378  {
379  VectorCopy(trace.endpos, view);
380  view[2] += (1.0f - trace.fraction) * 32.f;
381  // try another trace to this position, because a tunnel may have the ceiling
382  // close enogh that this is poking out
383 
384  CG_Trace(&trace, cg.refdef_current->vieworg, mins, maxs, view, cg.predictedPlayerState.clientNum, MASK_SOLID);
385  VectorCopy(trace.endpos, view);
386  }
387 
388  VectorCopy(view, cg.refdef_current->vieworg);
389 
390  // select pitch to look at focus point from vieword
391  VectorSubtract(focusPoint, cg.refdef_current->vieworg, focusPoint);
392  focusDist = sqrt(focusPoint[0] * focusPoint[0] + focusPoint[1] * focusPoint[1]);
393  if (focusDist < 1)
394  {
395  focusDist = 1; // should never happen
396  }
397  cg.refdefViewAngles[PITCH] = -180 / M_PI * atan2(focusPoint[2], focusDist);
398  cg.refdefViewAngles[YAW] -= cg_thirdPersonAngle.value;
399 }

424 void CG_KickAngles(void)
425 {
426  const vec3_t centerSpeed = { 2400, 2400, 2400 };
427  const float recoilCenterSpeed = 200;
428  const float recoilIgnoreCutoff = 15;
429  const float recoilMaxSpeed = 50;
430  const vec3_t maxKickAngles = { 10, 10, 10 };
431  float idealCenterSpeed, kickChange;
432  int i, frametime, t;
433  float ft;
434 #define STEP 20
435  char buf[32];
436 
437  // this code is frametime-dependant, so split it up into small chunks
438  //cg.kickAngles[PITCH] = 0;
439  cg.recoilPitchAngle = 0;
440  for (t = cg.frametime; t > 0; t -= STEP)
441  {
442  if (t > STEP)
443  {
444  frametime = STEP;
445  }
446  else
447  {
448  frametime = t;
449  }
450 
451  ft = frametime / 1000.f;
452 
453  // kickAngles is spring-centered
454  for (i = 0; i < 3; i++)
455  {
456  if (cg.kickAVel[i] != 0.f || cg.kickAngles[i] != 0.f)
457  {
458  // apply centering forces to kickAvel
459  if (cg.kickAngles[i] != 0.f && frametime)
460  {
461  idealCenterSpeed = -(2.0f * (cg.kickAngles[i] > 0) - 1.0f) * centerSpeed[i];
462  if (idealCenterSpeed != 0.f)
463  {
464  cg.kickAVel[i] += idealCenterSpeed * ft;
465  }
466  }
467  // add the kickAVel to the kickAngles
468  kickChange = cg.kickAVel[i] * ft;
469  if (cg.kickAngles[i] != 0.f && (cg.kickAngles[i] < 0) != (kickChange < 0)) // slower when returning to center
470  {
471  kickChange *= 0.06;
472  }
473  // check for crossing back over the center point
474  if (cg.kickAngles[i] == 0.f || ((cg.kickAngles[i] + kickChange) < 0) == (cg.kickAngles[i] < 0))
475  {
476  cg.kickAngles[i] += kickChange;
477  if (cg.kickAngles[i] == 0.f && frametime)
478  {
479  cg.kickAVel[i] = 0;
480  }
481  else if (Q_fabs(cg.kickAngles[i]) > maxKickAngles[i])
482  {
483  cg.kickAngles[i] = maxKickAngles[i] * ((2 * (cg.kickAngles[i] > 0)) - 1);
484  cg.kickAVel[i] = 0; // force Avel to return us to center rather than keep going outside range
485  }
486  }
487  else // about to cross, so just zero it out
488  {
489  cg.kickAngles[i] = 0;
490  cg.kickAVel[i] = 0;
491  }
492  }
493  }
494 
495  // recoil is added to input viewangles per frame
496  if (cg.recoilPitch != 0.f)
497  {
498  // apply max recoil
499  if (Q_fabs(cg.recoilPitch) > recoilMaxSpeed)
500  {
501  if (cg.recoilPitch > 0)
502  {
503  cg.recoilPitch = recoilMaxSpeed;
504  }
505  else
506  {
507  cg.recoilPitch = -recoilMaxSpeed;
508  }
509  }
510  // apply centering forces to kickAvel
511  if (frametime)
512  {
513  idealCenterSpeed = -(2.0f * (cg.recoilPitch > 0) - 1.0f) * recoilCenterSpeed * ft;
514  if (idealCenterSpeed != 0.f)
515  {
516  if (Q_fabs(idealCenterSpeed) < Q_fabs(cg.recoilPitch))
517  {
518  cg.recoilPitch += idealCenterSpeed;
519  }
520  else // back zero out
521  {
522  cg.recoilPitch = 0;
523  }
524  }
525  }
526  }
527  if (Q_fabs(cg.recoilPitch) > recoilIgnoreCutoff)
528  {
529  cg.recoilPitchAngle += cg.recoilPitch * ft;
530  }
531  }
532 
533  // only change cg_recoilPitch cvar when we need to
534  trap_Cvar_VariableStringBuffer("cg_recoilPitch", buf, sizeof(buf));
535 
536  if (atof(buf) != cg.recoilPitchAngle)
537  {
538  // encode the kick angles into a 24bit number, for sending to the client exe
539  trap_Cvar_Set("cg_recoilPitch", va("%f", cg.recoilPitchAngle));
540  }
541 }
