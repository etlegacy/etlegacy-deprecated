649 static uint8_t PredictPaeth(uint8_t a, uint8_t b, uint8_t c)
650 {
651  /*
652  * a == Left
653  * b == Up
654  * c == UpLeft
655  */
656 
657  uint8_t Pr;
658  int p;
659  int pa, pb, pc;
660 
661  p = ((int) a) + ((int) b) - ((int) c);
662  pa = abs(p - ((int) a));
663  pb = abs(p - ((int) b));
664  pc = abs(p - ((int) c));
665 
666  if ((pa <= pb) && (pa <= pc))
667  {
668  Pr = a;
669  }
670  else if (pb <= pc)
671  {
672  Pr = b;
673  }
674  else
675  {
676  Pr = c;
677  }
678 
679  return(Pr);

690 static qboolean UnfilterImage(uint8_t *DecompressedData,
691  uint32_t ImageHeight,
692  uint32_t BytesPerScanline,
693  uint32_t BytesPerPixel)
694 {
695  uint8_t *DecompPtr;
696  uint8_t FilterType;
697  uint8_t *PixelLeft, *PixelUp, *PixelUpLeft;
698  uint32_t w, h, p;
699 
700  // some zeros for the filters
701  uint8_t Zeros[8] = { 0, 0, 0, 0, 0, 0, 0, 0 };
702 
703  // input verification
704  if (!(DecompressedData && BytesPerPixel))
705  {
706  return qfalse;
707  }
708 
709  // ImageHeight and BytesPerScanline can be zero in small interlaced images.
710  if ((!ImageHeight) || (!BytesPerScanline))
711  {
712  return qtrue;
713  }
714 
715  // Set the pointer to the start of the decompressed Data.
716  DecompPtr = DecompressedData;
717 
718  // Un-filtering is done in place.
719 
720  // Go trough all scanlines.
721  for (h = 0; h < ImageHeight; h++)
722  {
723  // Every scanline starts with a FilterType byte.
724  FilterType = *DecompPtr;
725  DecompPtr++;
726 
727  // Left pixel of the first byte in a scanline is zero.
728  PixelLeft = Zeros;
729 
730  // Set PixelUp to previous line only if we are on the second line or above.
731  // Plus one byte for the FilterType
732  if (h > 0)
733  {
734  PixelUp = DecompPtr - (BytesPerScanline + 1);
735  }
736  else
737  {
738  PixelUp = Zeros;
739  }
740 
741  // The pixel left to the first pixel of the previous scanline is zero too.
742  PixelUpLeft = Zeros;
743 
744  // Cycle trough all pixels of the scanline.
745  for (w = 0; w < (BytesPerScanline / BytesPerPixel); w++)
746  {
747  // Cycle trough the bytes of the pixel.
748  for (p = 0; p < BytesPerPixel; p++)
749  {
750  switch (FilterType)
751  {
752  case PNG_FilterType_None:
753  {
754  // The byte is unfiltered.
755  break;
756  }
757 
758  case PNG_FilterType_Sub:
759  {
760  DecompPtr[p] += PixelLeft[p];
761 
762  break;
763  }
764 
765  case PNG_FilterType_Up:
766  {
767  DecompPtr[p] += PixelUp[p];
768 
769  break;
770  }
771 
772  case PNG_FilterType_Average:
773  {
774  DecompPtr[p] += ((uint8_t) ((((uint16_t) PixelLeft[p]) + ((uint16_t) PixelUp[p])) / 2));
775 
776  break;
777  }
778 
779  case PNG_FilterType_Paeth:
780  {
781  DecompPtr[p] += PredictPaeth(PixelLeft[p], PixelUp[p], PixelUpLeft[p]);
782 
783  break;
784  }
785 
786  default:
787  {
788  return qfalse;
789  }
790  }
791  }
792 
793  PixelLeft = DecompPtr;
794 
795  // We only have a upleft pixel if we are on the second line or above.
796  if (h > 0)
797  {
798  PixelUpLeft = DecompPtr - (BytesPerScanline + 1);
799  }
800 
801  // Skip to the next pixel.
802  DecompPtr += BytesPerPixel;
803 
804  // We only have a previous line if we are on the second line and above.
805  if (h > 0)
806  {
807  PixelUp = DecompPtr - (BytesPerScanline + 1);
808  }
809  }
810  }
811 
812  return qtrue;
813 }
