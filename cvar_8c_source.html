463  {
464  if (!(flags & CVAR_VM_CREATED))
465  {
466  var->flags &= ~CVAR_VM_CREATED;
467  }
468  }

471  if (flags & CVAR_VM_CREATED)
472  {
473  flags &= ~CVAR_VM_CREATED;
474  }
475  }
476 
477  // Make sure servers cannot mark engine-added variables as SERVER_CREATED
478  if (var->flags & CVAR_SERVER_CREATED)
479  {
480  if (!(flags & CVAR_SERVER_CREATED))
481  {
482  var->flags &= ~CVAR_SERVER_CREATED;
483  }
484  }
485  else
486  {
487  if (flags & CVAR_SERVER_CREATED)
488  {
489  flags &= ~CVAR_SERVER_CREATED;
490  }
491  }

494 
495  // only allow one non-empty reset string without a warning
496  if (!var->resetString[0])

523

527  }
528  }

531  // SERVERINFO get sent to clients
532  cvar_modifiedFlags |= flags;
533 
534  return var;
535  }
536

538  // allocate a new cvar
539  //
540 
541  // find a free cvar
542  for (index = 0; index < MAX_CVARS; index++)
543  {

589  cvar_modifiedFlags |= var->flags;
590

592  var->hashIndex = hash;
593

595  if (hashTable[hash])
596  {
597  hashTable[hash]->hashPrev = var;
598  }
599 
600  var->hashPrev = NULL;
601  hashTable[hash] = var;
602 
603  return var;
604 }

626  var_name = "BADNAME";
627  }
628 
629  var = Cvar_FindVar(var_name);
630  if (!var)
631  {
632  if (!value)
633  {
634  return NULL;
635  }
636  // create it
637  if (!force)
638  {
639  return Cvar_Get(var_name, value, CVAR_USER_CREATED);
640  }
641  else
642  {
643  return Cvar_Get(var_name, value, 0);
644  }
645  }
646 
647  if (!value)
648  {
649  value = var->resetString;
650  }
651 
652  value = Cvar_Validate(var, value, qtrue);
653 
654  if (var->flags & CVAR_USERINFO)
655  {
656  char *cleaned;
657 
658  cleaned = Cvar_ClearForeignCharacters(value);
659  if (strcmp(value, cleaned))

665 #endif
666  Com_Printf("Using %s instead of %s\n", cleaned, value);
667  return Cvar_Set2(var_name, cleaned, force);

672  {
673  if (!strcmp(value, var->string))
674  {
675  Z_Free(var->latchedString);
676  var->latchedString = NULL;
677  return var;
678  }
679 
680  if (!strcmp(value, var->latchedString))
681  {
682  return var;
683  }
684  }
685  else if (!strcmp(value, var->string))
686  {
687  return var;
688  }
689

691  cvar_modifiedFlags |= var->flags;
692 
693  if (!force)
694  {
695  // don't set unsafe variables when com_crashed is set
696  if ((var->flags & CVAR_UNSAFE) && com_crashed != NULL && com_crashed->integer)
697  {
698  Com_Printf("%s is unsafe. Check com_crashed.\n", var_name);

700  }
701 
702  if (var->flags & CVAR_ROM)
703  {
704  Com_Printf("%s is read only.\n", var_name);
705  return var;
706  }
707 
708  if (var->flags & CVAR_INIT)
709  {
710  Com_Printf("%s is write protected.\n", var_name);
711  return var;
712  }
713 
714  if ((var->flags & CVAR_CHEAT) && !cvar_cheats->integer)
715  {
716  Com_Printf("%s is cheat protected.\n", var_name);
717  return var;
718  }
719

721  {
722  Com_Printf("%s will be changed upon recompiling shaders.\n", var_name);
723  Cvar_Set("r_recompileShaders", "1");
724  }
725

727  {

729  {
730  if (strcmp(value, var->latchedString) == 0)
731  {
732  return var;
733  }
734  Z_Free(var->latchedString);
735  }
736  else
737  {
738  if (strcmp(value, var->string) == 0)
739  {
740  return var;
741  }
742  }
743

745  var->latchedString = CopyString(value);
746  var->modified = qtrue;
747  var->modificationCount++;

749  }
750  }
751  else
752  {

754  {
755  Z_Free(var->latchedString);
756  var->latchedString = NULL;
757  }
758  }
759 
760  if (!strcmp(value, var->string))
761  {
762  return var; // not changed
763 
764  }
765  var->modified = qtrue;
766  var->modificationCount++;
767 
768  Z_Free(var->string); // free the old value string
769 
770  var->string = CopyString(value);
771  var->value = (float)(atof(var->string));
772  var->integer = atoi(var->string);
773
